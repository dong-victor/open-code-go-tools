# Hub 跨设备同步功能 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 为 OCGT 添加三种部署形态的 Hub 跨设备同步功能，使多台设备的 token 使用统计数据可跨设备查看。

**Architecture:** 采用"本地优先，同步可选"架构。每台 OCGT 实例在内存中维护实时累加计数器（按 model/route/client 维度拆分 three time periods），定期向中央 Hub 推送汇总数据。Hub 有三种部署形态：A）内嵌在 OCGT 进程中的 Go HTTP 服务器；B）独立运行的 Go 二进制进程；C）Cloudflare Worker。Hub 通过 SSE 实时推送聚合数据给已连接的客户端。

**Tech Stack:** Go (标准库 net/http, encoding/json, sync), JavaScript (Cloudflare Worker), HTML/CSS/JS (前端), Chart.js (图表)

**前提条件:** 已通过 grill-with-docs 完成需求对齐，CONTEXT.md 术语表已建立。

```go
// internal/hub/counters.go
package hub

import (
    "encoding/json"
    "os"
    "path/filepath"
    "sync"
    "time"
)

const countersFileName = "sync-counters.json"

// NewSyncCounters 创建并初始化计数器，从快照恢复 allTime 和 month
func NewSyncCounters(dataDir string) (*SyncCounters, error) {
    sc := &SyncCounters{
        today:    newPeriodCounter(),
        month:    newPeriodCounter(),
        allTime:  newPeriodCounter(),
        dayKey:   time.Now().Format("2006-01-02"),
        monthKey: time.Now().Format("2006-01"),
    }
    // 从磁盘恢复快照
    snapshotPath := filepath.Join(dataDir, countersFileName)
    data, err := os.ReadFile(snapshotPath)
    if err == nil {
        var snap struct {
            AllTime  *periodCounter `json:"allTime"`
            Month    *periodCounter `json:"month"`
            MonthKey string         `json:"monthKey"`
        }
        if json.Unmarshal(data, &snap) == nil {
            sc.allTime = snap.AllTime
            if snap.MonthKey == sc.monthKey {
                sc.month = snap.Month
            }
        }
    }
    // 确保 counter maps 非 nil（快照可能为空）
    if sc.allTime == nil { sc.allTime = newPeriodCounter() }
    if sc.month == nil { sc.month = newPeriodCounter() }
    return sc, nil
}

func newPeriodCounter() *periodCounter {
    return &periodCounter{
        ByModel:  make(map[string]*dimCounter),
        ByRoute:  make(map[string]*dimCounter),
        ByClient: make(map[string]*dimCounter),
    }
}

// Add 累加一次请求的用量到所有三个时间段
func (sc *SyncCounters) Add(model, route, client string, inputTokens, outputTokens, cacheReadTokens, cacheCreateTokens int, cost float64) {
    sc.mu.Lock()
    defer sc.mu.Unlock()

    now := time.Now()
    todayKey := now.Format("2006-01-02")
    monthKey := now.Format("2006-01")

    // 检测日期变更，重置 today
    if todayKey != sc.dayKey {
        sc.today = newPeriodCounter()
        sc.dayKey = todayKey
    }
    // 检测月份变更，重置 month
    if monthKey != sc.monthKey {
        sc.month = newPeriodCounter()
        sc.monthKey = monthKey
    }

    sc.today.add(model, route, client, inputTokens, outputTokens, cacheReadTokens, cacheCreateTokens, cost)
    sc.month.add(model, route, client, inputTokens, outputTokens, cacheReadTokens, cacheCreateTokens, cost)
    sc.allTime.add(model, route, client, inputTokens, outputTokens, cacheReadTokens, cacheCreateTokens, cost)
}

func (pc *periodCounter) add(model, route, client string, inputTokens, outputTokens, cacheReadTokens, cacheCreateTokens int, cost float64) {
    total := int64(inputTokens + outputTokens + cacheReadTokens + cacheCreateTokens)
    pc.TotalTokens += total
    pc.EstimatedCost += cost
    pc.InputTokens += int64(inputTokens)
    pc.OutputTokens += int64(outputTokens)
    pc.CacheReadTokens += int64(cacheReadTokens)
    pc.CacheCreateTokens += int64(cacheCreateTokens)

    if model != "" {
        d := pc.ByModel[model]
        if d == nil { d = &dimCounter{}; pc.ByModel[model] = d }
        d.Tokens += total; d.Cost += cost; d.InputTokens += int64(inputTokens); d.OutputTokens += int64(outputTokens)
    }
    if route != "" {
        d := pc.ByRoute[route]
        if d == nil { d = &dimCounter{}; pc.ByRoute[route] = d }
        d.Tokens += total; d.Cost += cost; d.InputTokens += int64(inputTokens); d.OutputTokens += int64(outputTokens)
    }
    if client != "" {
        d := pc.ByClient[client]
        if d == nil { d = &dimCounter{}; pc.ByClient[client] = d }
        d.Tokens += total; d.Cost += cost; d.InputTokens += int64(inputTokens); d.OutputTokens += int64(outputTokens)
    }
}

// Snapshot 获取当前所有计数器的快照（用于推送）
func (sc *SyncCounters) Snapshot() SyncPayload {
    sc.mu.Lock()
    defer sc.mu.Unlock()
    // 确保日期/月份正确
    now := time.Now()
    if now.Format("2006-01-02") != sc.dayKey {
        sc.today = newPeriodCounter()
        sc.dayKey = now.Format("2006-01-02")
    }
    if now.Format("2006-01") != sc.monthKey {
        sc.month = newPeriodCounter()
        sc.monthKey = now.Format("2006-01")
    }
    return SyncPayload{
        UpdatedAt: now.UTC().Format(time.RFC3339),
        Today:     sc.today.snapshot(),
        Month:     sc.month.snapshot(),
        AllTime:   sc.allTime.snapshot(),
    }
}

func (pc *periodCounter) snapshot() PeriodStats {
    ps := PeriodStats{
        TotalTokens:       pc.TotalTokens,
        EstimatedCost:     pc.EstimatedCost,
        InputTokens:       pc.InputTokens,
        OutputTokens:      pc.OutputTokens,
        CacheReadTokens:   pc.CacheReadTokens,
        CacheCreateTokens: pc.CacheCreateTokens,
        ByModel:  make(map[string]DimStats, len(pc.ByModel)),
        ByRoute:  make(map[string]DimStats, len(pc.ByRoute)),
        ByClient: make(map[string]DimStats, len(pc.ByClient)),
    }
    for k, v := range pc.ByModel { ps.ByModel[k] = v.snapshot() }
    for k, v := range pc.ByRoute { ps.ByRoute[k] = v.snapshot() }
    for k, v := range pc.ByClient { ps.ByClient[k] = v.snapshot() }
    return ps
}

func (dc *dimCounter) snapshot() DimStats {
    return DimStats{Tokens: dc.Tokens, Cost: dc.Cost, InputTokens: dc.InputTokens, OutputTokens: dc.OutputTokens}
}

// HasData 检查是否有任何数据
func (sc *SyncCounters) HasData() bool {
    sc.mu.Lock()
    defer sc.mu.Unlock()
    return sc.today.TotalTokens > 0 || sc.month.TotalTokens > 0 || sc.allTime.TotalTokens > 0
}

// SaveSnapshot 持久化 allTime 和 month 到磁盘（用于进程重启恢复）
func (sc *SyncCounters) SaveSnapshot(dataDir string) error {
    sc.mu.Lock()
    snapshot := struct {
        AllTime  *periodCounter `json:"allTime"`
        Month    *periodCounter `json:"month"`
        MonthKey string         `json:"monthKey"`
    }{
        AllTime:  sc.allTime,
        Month:    sc.month,
        MonthKey: sc.monthKey,
    }
    sc.mu.Unlock()

    data, err := json.MarshalIndent(snapshot, "", "  ")
    if err != nil { return err }

    snapshotPath := filepath.Join(dataDir, countersFileName)
    if err := os.MkdirAll(filepath.Dir(snapshotPath), 0700); err != nil { return err }
    return os.WriteFile(snapshotPath, data, 0600)
}
```

在 `handler.go` 的 `addHistoryEntryWithUsageAndError` 函数中，`s.persistHistoryEntry(entry)` 之前插入：

```go
// 累加同步计数器
if s.HubCounters != nil {
    cost := 0.0 // cost 由 HubCounters 调用者从外部设置更合理，但为了解耦
    // 注意: pricing.EstimateCost 在 handler 中未导入。改用简化方式：
    // 在 hub 包中提供一个公开的 SetCostEstimator 来注册定价函数
    // 或者直接在 hub 包中硬编码一个简化版本。为了不引入循环依赖，
    // 从 hub 包外部注册一个 cost 函数。
    s.HubCounters.Add(entry.Model, entry.Route, usage.Client,
        usage.InputTokens, usage.OutputTokens, usage.CacheReadTokens, usage.CacheCreationTokens, cost)

```

**但是**，这里有一个问题：`hub` 包不能导入 `pricing` 包（避免循环依赖），且 `pricing.EstimateCost` 需要 model 名称。解决方案：在 `SyncCounters` 中支持设置一个可选的 cost 估算函数：

```go
// 在 counters.go 的 SyncCounters 中增加
type CostEstimator func(model string, inputTokens, outputTokens, cacheReadTokens, cacheCreateTokens int) float64

// 可选的 cost 估算器，由外部注册
var DefaultCostEstimator CostEstimator
```

然后在 handler.go 中注册：

```go
// 在 init() 或启动时注册
hub.DefaultCostEstimator = pricing.EstimateCost
```

调整 `periodCounter.add()` 使其在 `model != ""` 且 `DefaultCostEstimator != nil` 时自动计算 cost。

或者在 handler.go 中直接计算 cost 作为参数传入。这样最简单——调用方计算 cost：

```go
// handler.go addHistoryEntryWithUsageAndError 中

if s.HubCounters != nil && s.HubCounters.HasCostEstimator() {
    syncCost = s.HubCounters.EstimateCost(model, usage.InputTokens, usage.OutputTokens, usage.CacheReadTokens, usage.CacheCreationTokens)
}
```

**简化方案：** 不需要在 hub 包中引入 cost 计算，直接把 cost 作为 `parameters` 传给 `Add()`。这样 handler 调用时自己算 cost（或者暂时传 0，后续优化）：

```go

    // 简化：先不传 cost，后续可以加
    s.HubCounters.Add(entry.Model, entry.Route, usage.Client,
        usage.InputTokens, usage.OutputTokens, usage.CacheReadTokens, usage.CacheCreationTokens, 0)
}
```

Cost 计算可以在推送时通过 `pricing.EstimateCost` 估算，或者专门在 hub 包内维护一份简化版价格表。最简洁的方案：**在 hub 包内维护一份只读价格表**，避免跨包依赖。

```go
// EstimateCost 内置简化价格估算（不依赖 pricing 包）
func EstimateCost(model string, inputTokens, outputTokens, cacheReadTokens, cacheCreateTokens int) float64 {
    // 从内部价格表查找
    rate, ok := builtinRates[model]
    if !ok { rate = builtinRates["default"] }

    nonCacheInput := float64(inputTokens - cacheReadTokens)
    if nonCacheInput < 0 { nonCacheInput = 0 }
    cost := nonCacheInput*rate.InputRate +
        float64(outputTokens)*rate.OutputRate +
        float64(cacheReadTokens)*rate.CacheReadRate +
        float64(cacheCreateTokens)*rate.CacheWriteRate
    return cost
}

type modelRate struct {
    InputRate     float64
    OutputRate    float64
    CacheReadRate float64
    CacheWriteRate float64
}

// 常用模型价格（大致值，不需要精确到小数点后 6 位）
var builtinRates = map[string]modelRate{
    "default":             {InputRate: 3e-6, OutputRate: 15e-6, CacheReadRate: 0.3e-6, CacheWriteRate: 3e-6},
    "claude-sonnet-4-7":   {InputRate: 3e-6, OutputRate: 15e-6, CacheReadRate: 0.3e-6, CacheWriteRate: 3e-6},
    "deepseek-v4-flash":   {InputRate: 0.3e-6, OutputRate: 1.1e-6, CacheReadRate: 0.03e-6, CacheWriteRate: 0.3e-6},
    "deepseek-v4-pro":     {InputRate: 1.2e-6, OutputRate: 4e-6, CacheReadRate: 0.12e-6, CacheWriteRate: 1.2e-6},
    // 其他模型...
}
```

```go
    cost := EstimateCost(model, inputTokens, outputTokens, cacheReadTokens, cacheCreateTokens)
    // ... 原有累加逻辑 ...
}
```

```go
// internal/hub/server.go
package hub

import (
    "encoding/json"
    "fmt"
    "io"
    "log"
    "net"
    "net/http"
    "os"
    "path/filepath"
    "sync"
    "time"
)

const (
    DefaultHubPort    = 17321
    hubDataFileName   = "devices.json"
    sseHeartbeatSec   = 30
)

type Server struct {
    store        *HubStore
    storePath    string
    secret       string
    staleAfterMs int64
    sseClients   map[io.Writer]struct{}
    sseMu        sync.Mutex
    listener     net.Listener
    httpServer   *http.Server
    logger       *log.Logger
}

type ServerOption struct {
    Port         int
    Host         string
    Secret       string
    DataDir      string
    StaleAfterMs int64
    Logger       *log.Logger
}

func NewHubServer(opt ServerOption) (*Server, error) {
    if opt.Port == 0 { opt.Port = DefaultHubPort }
    if opt.Host == "" { opt.Host = "0.0.0.0" }
    if opt.DataDir == "" {
        home, err := os.UserHomeDir()
        if err != nil { return nil, fmt.Errorf("cannot determine home dir: %w", err) }
        opt.DataDir = filepath.Join(home, ".ocgt")
    }
    if opt.StaleAfterMs == 0 { opt.StaleAfterMs = DefaultStaleAfterMs }
    if opt.Logger == nil { opt.Logger = log.Default() }

    storePath := filepath.Join(opt.DataDir, hubDataFileName)
    store := loadStore(storePath)

    // 安全：无 secret 时只允许 loopback 连接
    bindHost := opt.Host
    if opt.Secret == "" && !isLoopback(opt.Host) {
        bindHost = "127.0.0.1"
        opt.Logger.Printf("[hub] no secret set, binding to %s only", bindHost)
    }

    s := &Server{
        store:        store,
        storePath:    storePath,
        secret:       opt.Secret,
        staleAfterMs: opt.StaleAfterMs,
        sseClients:   make(map[io.Writer]struct{}),
        logger:       opt.Logger,
    }

    mux := http.NewServeMux()
    mux.HandleFunc("/api/health", s.handleHealth)
    mux.HandleFunc("/api/stats", s.requireAuth(s.handleStats))
    mux.HandleFunc("/api/stats/stream", s.requireAuth(s.handleSSE))
    mux.HandleFunc("/api/devices", s.requireAuth(s.handleDevices))
    mux.HandleFunc("/api/ingest", s.requireAuth(s.handleIngest))

    s.httpServer = &http.Server{Handler: mux}

    addr := fmt.Sprintf("%s:%d", bindHost, opt.Port)
    ln, err := net.Listen("tcp", addr)
    if err != nil { return nil, fmt.Errorf("hub listen %s: %w", addr, err) }
    s.listener = ln

    return s, nil
}

func (s *Server) Start() error {
    s.logger.Printf("[hub] listening on http://%s", s.listener.Addr())
    return s.httpServer.Serve(s.listener)
}

func (s *Server) Addr() net.Addr { return s.listener.Addr() }

func (s *Server) Stop() error { return s.httpServer.Close() }

// ── 处理函数 ──

func (s *Server) handleHealth(w http.ResponseWriter, r *http.Request) {
    writeJSON(w, http.StatusOK, map[string]any{
        "ok": true, "role": "hub", "version": 1,
        "deviceCount": len(s.store.Devices),
        "secretRequired": s.secret != "",
        "now": time.Now().UTC().Format(time.RFC3339),
    })
}

func (s *Server) handleStats(w http.ResponseWriter, r *http.Request) {
    stats := s.aggregateStats()
    writeJSON(w, http.StatusOK, stats)
}

func (s *Server) handleSSE(w http.ResponseWriter, r *http.Request) {
    flusher, ok := w.(http.Flusher)
    if !ok { http.Error(w, "streaming not supported", http.StatusInternalServerError); return }

    w.Header().Set("Content-Type", "text/event-stream")
    w.Header().Set("Cache-Control", "no-cache")
    w.Header().Set("Connection", "keep-alive")
    w.Header().Set("X-Accel-Buffering", "no")

    // 发送初始快照
    stats := s.aggregateStats()
    s.writeSSE(w, "snapshot", stats)
    flusher.Flush()

    // 注册客户端
    s.sseMu.Lock()
    s.sseClients[w] = struct{}{}
    s.sseMu.Unlock()

    // 心跳
    ticker := time.NewTicker(sseHeartbeatSec * time.Second)
    defer ticker.Stop()

    done := r.Context().Done()
    for {
        select {
        case <-done:
            s.sseMu.Lock()
            delete(s.sseClients, w)
            s.sseMu.Unlock()
            return
        case <-ticker.C:
            fmt.Fprintf(w, ": hb\n\n")
            flusher.Flush()
        }
    }
}

func (s *Server) handleDevices(w http.ResponseWriter, r *http.Request) {
    if r.Method == http.MethodDelete {
        // DELETE /api/devices/{id}
        id := r.URL.Path[len("/api/devices/"):]
        s.storeMu.Lock()
        delete(s.store.Devices, id)
        s.persist()
        s.storeMu.Unlock()
        s.broadcast("delete")
        writeJSON(w, http.StatusOK, map[string]any{"ok": true, "deviceId": id})
        return
    }
    devices := make([]DeviceRecord, 0, len(s.store.Devices))
    for _, d := range s.store.Devices { devices = append(devices, d) }
    writeJSON(w, http.StatusOK, map[string]any{"devices": devices})
}

func (s *Server) handleIngest(w http.ResponseWriter, r *http.Request) {
    if r.Method != http.MethodPost {
        http.Error(w, "method not allowed", http.StatusMethodNotAllowed)
        return
    }
    body, err := io.ReadAll(io.LimitReader(r.Body, 1<<20))
    if err != nil { writeJSON(w, http.StatusBadRequest, map[string]string{"error": "bad_request"}); return }
    r.Body.Close()

    var payload SyncPayload
    if err := json.Unmarshal(body, &payload); err != nil {
        writeJSON(w, http.StatusBadRequest, map[string]string{"error": "bad_request"})
        return
    }
    if payload.DeviceID == "" {
        writeJSON(w, http.StatusBadRequest, map[string]string{"error": "deviceId_required"})
        return
    }

    record := DeviceRecord{
        DeviceID:    payload.DeviceID,
        DisplayName: payload.DisplayName,
        Hostname:    payload.Hostname,
        Platform:    payload.Platform,
        Version:     payload.Version,
        UpdatedAt:   payload.UpdatedAt,
        ReceivedAt:  time.Now().UTC().Format(time.RFC3339),
        Periods: map[string]PeriodStats{
            PeriodToday:   payload.Today,
            PeriodMonth:   payload.Month,
            PeriodAllTime: payload.AllTime,
        },
    }

    s.storeMu.Lock()
    // 如果已存在，合并（保留该设备已知的 trackedClients 等额外字段）
    existing, exists := s.store.Devices[payload.DeviceID]
    if exists {
        record.DisplayName = coalesce(record.DisplayName, existing.DisplayName)
        record.Hostname = coalesce(record.Hostname, existing.Hostname)
        record.Platform = coalesce(record.Platform, existing.Platform)
        // 保留历史 period 数据中本推送未携带的字段
        for _, p := range []string{PeriodToday, PeriodMonth, PeriodAllTime} {
            if _, has := record.Periods[p]; !has {
                if ep, ok := existing.Periods[p]; ok { record.Periods[p] = ep }
            }
        }
    }
    s.store.Devices[payload.DeviceID] = record
    s.persist()
    s.storeMu.Unlock()

    s.broadcast("ingest")
    writeJSON(w, http.StatusOK, map[string]any{
        "ok": true, "deviceId": payload.DeviceID,
        "stats": s.aggregateStats(),
    })
}

// ── 辅助方法 ──

func (s *Server) requireAuth(next http.HandlerFunc) http.HandlerFunc {
    if s.secret == "" { return next }
    return func(w http.ResponseWriter, r *http.Request) {
        auth := r.Header.Get("Authorization")
        if strings.HasPrefix(strings.ToLower(auth), "bearer ") && strings.TrimSpace(auth[7:]) == s.secret {
            next(w, r); return
        }
        if r.Header.Get("X-OCGT-Secret") == s.secret {
            next(w, r); return
        }
        writeJSON(w, http.StatusUnauthorized, map[string]string{"error": "unauthorized"})
    }
}

func (s *Server) aggregateStats() map[string]any {
    s.storeMu.Lock()
    defer s.storeMu.Unlock()

    now := time.Now()
    nowMs := now.UnixMilli()
    agg := map[string]any{
        "updatedAt": now.UTC().Format(time.RFC3339),
        "devices":   []any{},
    }
    for _, p := range []string{PeriodToday, PeriodMonth, PeriodAllTime} {
        agg[p] = PeriodStats{
            ByModel:  make(map[string]DimStats),
            ByRoute:  make(map[string]DimStats),
            ByClient: make(map[string]DimStats),
        }
    }

    deviceList := make([]map[string]any, 0)
    for _, dev := range s.store.Devices {
        receivedMs := parseTimeMs(dev.ReceivedAt)
        stale := receivedMs > 0 && nowMs-receivedMs > s.staleAfterMs

        d := map[string]any{
            "deviceId": dev.DeviceID, "displayName": dev.DisplayName,
            "hostname": dev.Hostname, "platform": dev.Platform,
            "version": dev.Version, "updatedAt": dev.UpdatedAt,
            "receivedAt": dev.ReceivedAt, "stale": stale,
            "periods": dev.Periods,
        }
        deviceList = append(deviceList, d)

        // 累加到汇总
        for _, p := range []string{PeriodToday, PeriodMonth, PeriodAllTime} {
            if ps, ok := dev.Periods[p]; ok {
                target := agg[p].(PeriodStats)
                target.TotalTokens += ps.TotalTokens
                target.EstimatedCost += ps.EstimatedCost
                target.InputTokens += ps.InputTokens
                target.OutputTokens += ps.OutputTokens
                target.CacheReadTokens += ps.CacheReadTokens
                target.CacheCreateTokens += ps.CacheCreateTokens
                mergeDimMap(target.ByModel, ps.ByModel)
                mergeDimMap(target.ByRoute, ps.ByRoute)
                mergeDimMap(target.ByClient, ps.ByClient)
                agg[p] = target
            }
        }
    }

    agg["devices"] = deviceList
    return agg
}

// ── store 持久化 ──

func loadStore(path string) *HubStore {
    data, err := os.ReadFile(path)
    if err != nil {
        return &HubStore{Version: 1, Devices: make(map[string]DeviceRecord)}
    }
    var store HubStore
    if json.Unmarshal(data, &store) != nil || store.Devices == nil {
        return &HubStore{Version: 1, Devices: make(map[string]DeviceRecord)}
    }
    return &store
}

func (s *Server) persist() {
    s.store.SavedAt = time.Now().UTC().Format(time.RFC3339)
    data, err := json.MarshalIndent(s.store, "", "  ")
    if err != nil { return }
    os.MkdirAll(filepath.Dir(s.storePath), 0700)
    os.WriteFile(s.storePath, data, 0600)
}

// ── SSE ──

func (s *Server) writeSSE(w io.Writer, event string, data any) {
    d, _ := json.Marshal(data)
    fmt.Fprintf(w, "event: %s\ndata: %s\n\n", event, d)
}

func (s *Server) broadcast(reason string) {
    s.sseMu.Lock()
    clients := make([]io.Writer, 0, len(s.sseClients))
    for c := range s.sseClients { clients = append(clients, c) }
    s.sseMu.Unlock()
    if len(clients) == 0 { return }

    stats := s.aggregateStats()
    for _, c := range clients {
        s.writeSSE(c, "stats", map[string]any{
            "type": "stats", "reason": reason, "stats": stats,
            "at": time.Now().UTC().Format(time.RFC3339),
        })
        if flusher, ok := c.(http.Flusher); ok { flusher.Flush() }
    }
}

// ── 工具函数 ──

func writeJSON(w http.ResponseWriter, status int, v any) {
    w.Header().Set("Content-Type", "application/json")
    w.WriteHeader(status)
    json.NewEncoder(w).Encode(v)
}

func isLoopback(host string) bool {
    switch host { case "127.0.0.1", "localhost", "::1", "": return true }
    return false
}

func coalesce(a, b string) string { if a != "" { return a }; return b }

func parseTimeMs(s string) int64 {
    t, err := time.Parse(time.RFC3339, s)
    if err != nil { return 0 }
    return t.UnixMilli()
}

func mergeDimMap(target, source map[string]DimStats) {
    for k, v := range source {
        if existing, ok := target[k]; ok {
            existing.Tokens += v.Tokens
            existing.Cost += v.Cost
            existing.InputTokens += v.InputTokens
            existing.OutputTokens += v.OutputTokens
            target[k] = existing
        } else {
            target[k] = v
        }
    }
}
```

```go
// internal/hub/client.go
package hub

import (
    "bytes"
    "context"
    "encoding/json"
    "fmt"
    "io"
    "log"
    "net/http"
    "os"
    "runtime"
    "time"
)

type Client struct {
    config     Config
    counters   *SyncCounters
    deviceID   string
    version    string
    httpClient *http.Client
    ctx        context.Context
    cancel     context.CancelFunc
    pushTicker *time.Ticker
    logger     *log.Logger
    dataDir    string

    // 来自 Hub 的远程设备数据
    remoteStats atomic.Value // 存储 map[string]any
}

func NewClient(cfg Config, counters *SyncCounters, version, dataDir string) (*Client, error) {
    if cfg.PushIntervalSec < MinPushIntervalSec { cfg.PushIntervalSec = DefaultPushIntervalSec }
    if cfg.PushIntervalSec > MaxPushIntervalSec { cfg.PushIntervalSec = MaxPushIntervalSec }

    // 获取或生成本机设备 ID
    deviceID := loadOrCreateDeviceID(dataDir)

    ctx, cancel := context.WithCancel(context.Background())
    c := &Client{
        config:     cfg,
        counters:   counters,
        deviceID:   deviceID,
        version:    version,
        httpClient: &http.Client{Timeout: 30 * time.Second},
        ctx:        ctx,
        cancel:     cancel,
        logger:     log.Default(),
        dataDir:    dataDir,
    }
    c.remoteStats.Store(map[string]any{})

    return c, nil
}

func (c *Client) Start() {
    if !c.config.Enabled || c.config.HubURL == "" { return }

    c.pushTicker = time.NewTicker(time.Duration(c.config.PushIntervalSec) * time.Second)

    // 立即执行一次推送
    go c.pushOnce()
    // 定时推送
    go func() {
        for {
            select {
            case <-c.ctx.Done():
                c.pushOnce() // 退出前最后推一次
                return
            case <-c.pushTicker.C:
                c.pushOnce()
            }
        }
    }()

    // 启动 SSE 接收（如果 Hub 支持）
    go c.listenSSE()
}

func (c *Client) Stop() {
    c.cancel()
    if c.pushTicker != nil { c.pushTicker.Stop() }
}

func (c *Client) pushOnce() {
    if !c.config.Enabled || c.config.HubURL == "" { return }

    payload := c.counters.Snapshot()
    if !c.counters.HasData() { return } // 无数据跳过推送

    payload.DeviceID = c.deviceID
    payload.DisplayName = c.config.DeviceName
    payload.Hostname, _ = os.Hostname()
    payload.Platform = runtime.GOOS + "-" + runtime.GOARCH
    payload.Version = c.version

    body, _ := json.Marshal(payload)
    url := c.config.HubURL + "/api/ingest"

    req, err := http.NewRequest("POST", url, bytes.NewReader(body))
    if err != nil { return }
    req.Header.Set("Content-Type", "application/json")
    if c.config.Secret != "" {
        req.Header.Set("Authorization", "Bearer "+c.config.Secret)
    }

    resp, err := c.httpClient.Do(req)
    if err != nil {
        c.logger.Printf("[sync] push failed: %v", err)
        return
    }
    defer resp.Body.Close()

    if resp.StatusCode != http.StatusOK {
        c.logger.Printf("[sync] push returned %d", resp.StatusCode)
        return
    }

    // 保存快照（推送成功后才保存，确保不丢失已推送的数据）
    c.counters.SaveSnapshot(c.dataDir)
}

func (c *Client) listenSSE() {
    if c.config.HubURL == "" { return }

    url := c.config.HubURL + "/api/stats/stream"
    for {
        select {
        case <-c.ctx.Done(): return
        default:
        }

        req, err := http.NewRequest("GET", url, nil)
        if err != nil { time.Sleep(30 * time.Second); continue }
        if c.config.Secret != "" {
            req.Header.Set("Authorization", "Bearer "+c.config.Secret)
        }

        resp, err := c.httpClient.Do(req)
        if err != nil {
            time.Sleep(30 * time.Second)
            continue
        }

        reader := bufio.NewReader(resp.Body)
        for {
            select {
            case <-c.ctx.Done():
                resp.Body.Close()
                return
            default:
            }

            line, err := reader.ReadString('\n')
            if err != nil {
                resp.Body.Close()
                break
            }

            if strings.HasPrefix(line, "data: ") {
                var event struct {
                    Type   string         `json:"type"`
                    Reason string         `json:"reason"`
                    Stats  map[string]any `json:"stats"`
                }
                if json.Unmarshal([]byte(line[6:]), &event) == nil && event.Stats != nil {
                    c.remoteStats.Store(event.Stats)
                }
            }
        }
        time.Sleep(5 * time.Second) // 重连等待
    }
}

func (c *Client) RemoteStats() map[string]any {
    v := c.remoteStats.Load()
    if v == nil { return nil }
    return v.(map[string]any)
}

func loadOrCreateDeviceID(dataDir string) string {
    idPath := filepath.Join(dataDir, "device-id")
    if data, err := os.ReadFile(idPath); err == nil && len(data) > 0 {
        return strings.TrimSpace(string(data))
    }
    id := fmt.Sprintf("ocgt-%s", uuid())
    os.MkdirAll(dataDir, 0700)
    os.WriteFile(idPath, []byte(id), 0600)
    return id
}

func uuid() string {
    b := make([]byte, 16)
    rand.Read(b)
    return fmt.Sprintf("%x-%x-%x-%x-%x", b[0:4], b[4:6], b[6:8], b[8:10], b[10:])
}
```

```javascript
const CORS_HEADERS = {
  'access-control-allow-origin': '*',
  'access-control-allow-methods': 'GET,POST,DELETE,OPTIONS',
  'access-control-allow-headers': 'authorization,content-type,x-ocgt-secret'
};

// 与 Go 端一致的数据聚合逻辑（简化版）
function aggregateDevices(devices, staleAfterMs, now) {
  const periods = ['today', 'month', 'allTime'];
  const agg = { updatedAt: new Date().toISOString(), devices: [], today: emptyPeriod(), month: emptyPeriod(), allTime: emptyPeriod() };

  for (const dev of devices) {
    const age = now - new Date(dev.receivedAt || dev.updatedAt).getTime();
    const stale = staleAfterMs > 0 && age > staleAfterMs;
    agg.devices.push({ ...dev, stale, ageMs: age });

    for (const p of periods) {
      const ps = dev.periods?.[p];
      if (!ps) continue;
      const target = agg[p];
      target.totalTokens += ps.totalTokens || 0;
      target.estimatedCost += ps.estimatedCost || 0;
      target.inputTokens += ps.inputTokens || 0;
      target.outputTokens += ps.outputTokens || 0;
      mergeDimMap(target.byModel, ps.byModel);
      mergeDimMap(target.byRoute, ps.byRoute);
      mergeDimMap(target.byClient, ps.byClient);
    }
  }
  return agg;
}

function emptyPeriod() {
  return { totalTokens: 0, estimatedCost: 0, inputTokens: 0, outputTokens: 0, cacheReadTokens: 0, cacheCreateTokens: 0, byModel: {}, byRoute: {}, byClient: {} };
}

function mergeDimMap(target, source) {
  if (!source) return;
  for (const [k, v] of Object.entries(source)) {
    if (target[k]) {
      target[k].tokens += v.tokens;
      target[k].cost += v.cost;
    } else {
      target[k] = { ...v };
    }
  }
}
```
